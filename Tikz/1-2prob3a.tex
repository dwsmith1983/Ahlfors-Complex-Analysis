\begin{document}
\begin{tikzpicture}
  % drawing unit circle and grid
  \draw[help lines, step = 0.25cm] (-1cm, -1cm) grid (1cm, 1cm);
  \draw[samples = 100, domain = 0:360]
  plot({cos(\x)}, {sin(\x)});
  % fill in roots of unit
  \foreach \k in{0, 2, 4, 6, 8}{
    \draw[thin] (0, 0) -- plot({cos(\k*180/5)}, {sin(\k*180/5)});
    \fill[red] plot({cos(\k*180/5)}, {sin(\k*180/5)})
    circle[radius = 0.025cm];
  }
\end{tikzpicture}
